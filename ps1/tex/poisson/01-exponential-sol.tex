\begin{answer}
    \begin{align*}
        b(y) &= \frac{1}{y!} \\
        \eta &= \log \lambda \\
        T(y) &= y \\
        a(\eta) &= \lambda = e^\eta
    \end{align*}
    Therefore it is in the exponential family as shown:
    $$
        p(y; \eta) = \frac{1}{y!} \exp(y \log \lambda - \lambda) = \frac{1}{y!} \exp(\eta T(y) - e^\eta)
    $$
\end{answer}
